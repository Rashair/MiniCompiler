			num = i;

				isPrime = true;
			}
			else if((num/2)*2 != num && (num/3)*3 != num)
			{
				val = 5;
				while(val * val <= num && val > 4)
				{
					val2 = val + 2;
					if ((num/val) * val == val || (num/val2) * val2 == val2)
					{
						val = 4;
					}					
					else
					{
						val = val + 6;
					}
				}

			}
