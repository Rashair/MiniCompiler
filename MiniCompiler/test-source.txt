program {
    int x;
    double d;
    bool b;
    
    read x;
    read d;
    read b;
    
    write x; write "\n";
    write d; write "\n";
    write b; write "\n";

    read b;
    read d;
    read x;

    write b; write "\n";
    write d; write "\n";
    write x; write "\n";
}
